### 智能音乐管理器

```javascript
class SmartMusicManager {
  constructor() {
    this.audio = new Audio();
    this.currentScene = 'default';
    this.volume = 0.25;
    this.musicLibrary = musicLibrary;
  }

  /**
   * 播放场景音乐
   * 优先使用经典音乐，如果不可用则使用AI音乐
   */
  playScene(scene) {
    const sceneConfig = this.getSceneConfig(scene);

    // 优先尝试经典音乐
    if (sceneConfig.classical && this.hasClassicalMusic(sceneConfig.classical)) {
      this.playClassical(sceneConfig.classical);
      console.log(`🎼 [经典音乐] ${sceneConfig.classical.title}`);
    }
    // 回退到AI音乐
    else if (sceneConfig.ai) {
      this.playAI(sceneConfig.ai);
      console.log(`🎵 [AI音乐] ${scene}`);
    }
    // 默认音乐
    else {
      this.playDefault();
    }
  }

  /**
   * 检查是否有经典音乐文件
   */
  hasClassicalMusic(classicalConfig) {
    // 检查文件是否存在
    return fetch(classicalConfig.file, { method: 'HEAD' })
      .then(response => response.ok)
      .catch(() => false);
  }

  /**
   * 获取场景配置
   */
  getSceneConfig(scene) {
    const sceneMap = {
      'buffett_lunch': {
        classical: this.musicLibrary.classical.buffett_lunch,
        ai: this.musicLibrary.ai.default, // 回退
        mood: 'elegant'
      },

      'bridge_card': {
        classical: this.musicLibrary.classical.bridge_card, // 可选
        ai: this.musicLibrary.ai.jazz_bridge, // 主要用这个
        mood: 'sophisticated'
      },

      'coca_cola': {
        classical: null,
        ai: this.musicLibrary.ai.coca_cola,
        mood: 'cheerful'
      },

      // ... 其他场景
    };

    return sceneMap[scene] || sceneMap['default'];
  }

  /**
   * 播放经典音乐
   */
  playClassical(config) {
    this.audio.src = config.file;
    this.audio.play();
    this.currentScene = 'classical';
  }

  /**
   * 播放AI音乐
   */
  playAI(config) {
    const randomFile = config.files[Math.floor(Math.random() * config.files.length)];
    this.audio.src = randomFile;
    this.audio.play();
    this.audio.loop = true; // AI音乐循环播放
    this.currentScene = 'ai';
  }

  playDefault() {
    this.playAI(this.musicLibrary.ai.default);
  }
}
```
